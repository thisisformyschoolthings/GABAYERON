<script type="text/javascript">

    function checkFavoriteAnswer() {
        const userAnswer = document.favoriteQuiz.answer.value.trim();


        const resultDiv = document.getElementById("favoriteResult");
        resultDiv.style.display = 'block';

    
        if (userAnswer === "G6PD Deficiency") {
            resultDiv.textContent = "Yep! That's why I can't have soy!";
            resultDiv.className = 'result correct';
        } else {
            resultDiv.textContent = "Noopee!";
            resultDiv.className = 'result incorrect';
        }
    }


    function checkLeastFavoriteAnswer() {
        const userAnswer = document.leastFavoriteQuiz.answer.value.trim();

        const resultDiv = document.getElementById("leastFavoriteResult");
        resultDiv.style.display = 'block';

   
        if (userAnswer === "Cereal") {
            resultDiv.textContent = "Oh my gosh, I love cereal!";
            resultDiv.className = 'result correct';
        } else {
            resultDiv.textContent = "Nu uh! Yuck!";
            resultDiv.className = 'result incorrect';
        }
    }
</script>
